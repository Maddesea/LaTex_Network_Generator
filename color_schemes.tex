% Load a color scheme
% Usage: \loadColorScheme{scheme_name}
\newcommand{\loadColorScheme}[1]{
    \IfStrEq{#1}{default}{\colorSchemeDefault}{}
    \IfStrEq{#1}{dark}{\colorSchemeDark}{}
    \IfStrEq{#1}{colorblind}{\colorSchemeColorblind}{}
    \IfStrEq{#1}{monochrome}{\colorSchemeMonochrome}{}
    \IfStrEq{#1}{high-contrast}{\colorSchemeHighContrast}{}
    \IfStrEq{#1}{pastel}{\colorSchemePastel}{}
    \IfStrEq{#1}{corporate}{\colorSchemeCorporate}{}
    \IfStrEq{#1}{neon}{\colorSchemeNeon}{}
    \IfStrEq{#1}{earth}{\colorSchemeEarth}{}
    \IfStrEq{#1}{ocean}{\colorSchemeOcean}{}
}

% ============================================================================
% THEME SWITCHER
% ============================================================================

% Set diagram theme (affects multiple properties)
% Usage: \setDiagramTheme{theme_name}
\newcommand{\setDiagramTheme}[1]{
    \IfStrEq{#1}{light}{
        \loadColorScheme{default}
        \def\bgColor{white}
        \def\textColor{black}
    }{}
    \IfStrEq{#1}{dark}{
        \loadColorScheme{dark}
        \def\bgColor{black}
        \def\textColor{white}
    }{}
    \IfStrEq{#1}{print}{
        \loadColorScheme{high-contrast}
        \def\bgColor{white}
        \def\textColor{black}
    }{}
    \IfStrEq{#1}{presentation}{
        \loadColorScheme{high-contrast}
        \def\bgColor{white}
        \def\textColor{black}
    }{}
}

% ============================================================================
% COLOR UTILITIES
% ============================================================================

% Lighten a color
% Usage: \lightenColor{color}{amount}{new_name}
\newcommand{\lightenColor}[3]{
    \colorlet{#3}{#1!#2!white}
}

% Darken a color
% Usage: \darkenColor{color}{amount}{new_name}
\newcommand{\darkenColor}[3]{
    \colorlet{#3}{#1!#2!black}
}

% Blend two colors
% Usage: \blendColors{color1}{color2}{ratio}{new_name}
\newcommand{\blendColors}[4]{
    \colorlet{#4}{#1!#3!#2}
}

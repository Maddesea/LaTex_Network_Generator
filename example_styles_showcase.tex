\section*{Example 1: Premium Gradient Styles (Standard Color Scheme)}

% Load standard color scheme (default)
\loadColorScheme{standard}

\begin{center}
\begin{tikzpicture}[scale=0.8, transform shape]

    % Premium gradient nodes
    \node[server premium] (web) at (0,0) {Web Server\\192.168.1.10};
    \node[client premium] (client1) at (-4,-3) {Client PC\\192.168.1.100};
    \node[router premium] (router) at (0,-3) {Router\\192.168.1.1};
    \node[firewall premium] (fw) at (4,-3) {Firewall\\192.168.1.254};
    \node[switch premium] (switch) at (0,-6) {Core Switch};

    % Add badges to nodes
    \nodeBadge{web}{linux badge}{north east}{L}
    \nodeBadge{web}{online badge}{north west}{\checkmark}
    \nodeBadge{client1}{windows badge}{north east}{W}
    \nodeBadge{client1}{warning badge}{north west}{!}
    \nodeBadge{fw}{online badge}{north east}{\checkmark}

    % Connections
    \draw[encrypted conn] (client1) -- (router);
    \draw[normal conn] (router) -- (web);
    \draw[normal conn] (router) -- (fw);
    \draw[normal conn] (switch) -- (router);

\end{tikzpicture}
\end{center}

\section*{Available Color Schemes and Usage}

\begin{itemize}
    \item \texttt{\textbackslash loadColorScheme\{standard\}} - Default vibrant colors
    \item \texttt{\textbackslash loadColorScheme\{dark\}} - Dark theme with bright accents
    \item \texttt{\textbackslash loadColorScheme\{colorblind\}} - Deuteranopia/protanopia safe
    \item \texttt{\textbackslash loadColorScheme\{tritanopia\}} - Tritanopia safe
    \item \texttt{\textbackslash loadColorScheme\{monochrome\}} - Grayscale for printing
    \item \texttt{\textbackslash loadColorScheme\{high-contrast\}} - High contrast for accessibility
\end{itemize}

\subsection*{Node Style Variants}

\begin{itemize}
    \item Standard: \texttt{server}, \texttt{client}, \texttt{router}, \texttt{firewall}, \texttt{switch}, \texttt{cloud}
    \item Premium (with gradients): \texttt{server premium}, \texttt{client premium}, etc.
    \item Colorblind (with patterns): \texttt{server colorblind}, \texttt{client colorblind}, etc.
\end{itemize}

\subsection*{Badge Support}

\begin{itemize}
    \item OS Badges: \texttt{windows badge}, \texttt{linux badge}, \texttt{macos badge}
    \item Status Badges: \texttt{online badge}, \texttt{offline badge}, \texttt{warning badge}, \texttt{critical badge}
    \item Usage: \texttt{\textbackslash nodeBadge\{node-name\}\{badge-style\}\{position\}\{text\}}
    \item Positions: \texttt{north east}, \texttt{north west}, \texttt{south east}, \texttt{south west}
\end{itemize}
